⚙️ JollyExplorers Scrum OS — First-time Setup

Connect your Google Sheet to store tasks, standups, and metrics. Takes about 5 minutes. Do this once.

1 Create a Google Sheet

Go to sheets.google.com and create a new blank spreadsheet. Name it "JollyExplorers Scrum".

2 Open Apps Script

In your Sheet, click Extensions → Apps Script. Delete all existing code, paste the script below, then click Save (Ctrl+S).

function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    if (e.parameter.action === 'init') { setupSheets(ss); return json({ success: true }); }
    const sheet = e.parameter.sheet;
    const ws = ss.getSheetByName(sheet);
    if (!ws) return json({ error: 'Sheet not found: ' + sheet });
    const values = ws.getDataRange().getValues();
    return json({ data: values });
  } catch(err) { return json({ error: err.toString() }); }
}

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const ws = ss.getSheetByName(payload.sheet);
    if (payload.action === 'init') { setupSheets(ss); return json({ success: true }); }
    if (!ws) return json({ error: 'Sheet not found: ' + payload.sheet });
    if (payload.action === 'append') { ws.appendRow(payload.row); }
    else if (payload.action === 'update_cell') { ws.getRange(payload.row, payload.col).setValue(payload.value); }
    else if (payload.action === 'delete_row') {
      const data = ws.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === String(payload.id)) { ws.deleteRow(i + 1); break; }
      }
    }
    return json({ success: true });
  } catch(err) { return json({ error: err.toString() }); }
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function getWeek(d) {
  const onejan = new Date(d.getFullYear(), 0, 1);
  return Math.ceil((((d - onejan) / 86400000) + onejan.getDay() + 1) / 7);
}
